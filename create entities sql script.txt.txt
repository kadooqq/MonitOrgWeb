CREATE TABLE vendors (vendor_name VARCHAR, vendor_shortName VARCHAR, vendor_region VARCHAR, vendor_id BIGSERIAL PRIMARY KEY);
CREATE TABLE products (vendor_id BIGINT REFERENCES vendors(vendor_id), product_name VARCHAR, product_seamType VARCHAR, product_subType VARCHAR, product_document VARCHAR, product_diameterMin DOUBLE PRECISION, product_diameterMax DOUBLE PRECISION, product_width DOUBLE PRECISION, product_height DOUBLE PRECISION, product_thicknessMin DOUBLE PRECISION, product_thicknessMax DOUBLE PRECISION, product_steelMark VARCHAR, product_price DOUBLE PRECISION, product_status VARCHAR, product_id BIGSERIAL PRIMARY KEY);
